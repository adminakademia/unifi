
## Instalacja zależności
sudo apt update && sudo apt install -y podman uidmap slirp4netns iptables curl wget ca-certificates sudo

## Weryfikacja instalacji aplikacji Podman (musi być w wersji conajmniej 4.3.1)
podman --version


----------------- Dla Debian/Ubuntu ------------------------------------

## Pobieramy instalator UniFi OS Server 
sudo mkdir -p /usr/local/src/unifi-os-server
cd /usr/local/src/unifi-os-server

sudo curl -LO https://fw-download.ubnt.com/data/unifi-os-server/8b93-linux-x64-4.2.23-158fa00b-6b2c-4cd8-94ea-e92bc4a81369.23-x64
sudo chmod +x ./8b93-linux-x64-4.2.23-158fa00b-6b2c-4cd8-94ea-e92bc4a81369.23-x64

## Instalujemy UniFi OS Server
sudo ./8b93-linux-x64-4.2.23-158fa00b-6b2c-4cd8-94ea-e92bc4a81369.23-x64 


----------------- Dla Raspberry PI OS ------------------------------------

## Pobieramy instalator UniFi OS Server

sudo cd /usr/local/src/unifi-os-server
